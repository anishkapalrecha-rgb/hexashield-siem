def init_db():
    # If you get "no such column" errors, delete the .db file and restart
    with sqlite3.connect(DB_PATH) as conn:
        conn.execute("""
            CREATE TABLE IF NOT EXISTS alerts(
                id INTEGER PRIMARY KEY AUTOINCREMENT,

                type TEXT, message TEXT, ip TEXT,
                country TEXT, severity TEXT, protocol TEXT
            )
        """)

# ---------------- ENGINES ----------------
geo_cache = {}
def get_geo(ip):
    if ip in geo_cache: return geo_cache[ip]
    try:
        # Fetching Country only to keep it fast
        r = requests.get(f"http://ip-api.com/json/{ip}?fields=country", timeout=1).json()
        country = r.get('country', 'Local/Unknown')
        geo_cache[ip] = country
        return country
    except: return "Discovery"

def send_mail(t, ip, loc):
    global target_email
    if not target_email or not SENDER_PASSWORD: return
    try:
        msg = MIMEMultipart()
        msg['From'] = SENDER_EMAIL
        msg['To'] = target_email
        msg['Subject'] = f"🚨 HexaShield ALERT: {t} Detected"
       
        body = f"Intrusion Detection System Alert\n\nType: {t}\nSource IP: {ip}\nLocation: {loc}\n\nCheck your dashboard for details."
        msg.attach(MIMEText(body, 'plain'))

        s = smtplib.SMTP("smtp.gmail.com", 587)
        s.starttls()
        s.login(SENDER_EMAIL, SENDER_PASSWORD)
        s.send_message(msg)
        s.quit()
        print(f"Alert sent to {target_email}")
    except Exception as e:
        print(f"Mail Error: {e}")

def log_event(t, msg, ip, proto):
    sev = "HIGH" if any(x in t for x in ["Scan", "Attack", "Flood"]) else "MEDIUM"
    loc = get_geo(ip)
    with sqlite3.connect(DB_PATH) as conn:
        conn.execute("INSERT INTO alerts (type, message, ip, country, severity, protocol) VALUES (?,?,?,?,?,?)",
                    (t, msg, ip, loc, sev, proto))
   
    # Send email ONLY for High severity alerts
    if sev == "HIGH":
        threading.Thread(target=send_mail, args=(t, ip, loc)).start()

def packet_hunter(pkt):
    if not pkt.haslayer(IP): return
    src = pkt[IP].src
    if pkt.haslayer(TCP):
        f = pkt[TCP].flags
        p = pkt[TCP].dport
        if f == 0: log_event("NULL Scan", f"Target Port {p}", src, "TCP")
        elif f & 0x29: log_event("XMAS Scan", f"Target Port {p}", src, "TCP")
        elif p in [80, 443]: log_event("Web Traffic", "HTTP/S Protocol", src, "WEB")
        else: log_event("TCP Traffic", f"Port {p}", src, "TCP")
    elif pkt.haslayer(UDP): log_event("UDP Data", f"Port {pkt[UDP].dport}", src, "UDP")
    elif pkt.haslayer(ICMP): log_event("ICMP Ping", "Echo Request", src, "ICMP")

# Sniffer Thread
threading.Thread(target=lambda: sniff(prn=packet_hunter, store=False), daemon=True).start()

@app.route('/')
def index():
    if 'user' not in session: return redirect('/login')
    with sqlite3.connect(DB_PATH) as conn:
        logs = conn.execute("SELECT * FROM alerts ORDER BY time DESC LIMIT 15").fetchall()
    return render_template_string(DASHBOARD_HTML, logs=logs, email=target_email)

@app.route('/api/stats')
def stats():
    with sqlite3.connect(DB_PATH) as conn:
        data = dict(conn.execute("SELECT protocol, COUNT(*) FROM alerts GROUP BY protocol").fetchall())
    return jsonify(data)

@app.route('/export/<fmt>')
def export(fmt):
    with sqlite3.connect(DB_PATH) as conn:
        data = conn.execute("SELECT time, type, ip, country, severity FROM alerts").fetchall()
    path = f"report.{fmt}"
    if fmt == 'csv':
        with open(path, 'w') as f: csv.writer(f).writerows([["Time","Type","IP","Loc","Severity"]]+list(data))
    else:
        doc = SimpleDocTemplate(path, pagesize=letter)
        t = Table([["Time","Type","IP","Loc","Severity"]]+[list(x) for x in data])
        t.setStyle(TableStyle([('BACKGROUND',(0,0),(-1,0),colors.black),('TEXTCOLOR',(0,0),(-1,0),colors.white),('GRID',(0,0),(-1,-1),0.5,colors.grey)]))
        doc.build([Paragraph("HexaShield Audit Report", getSampleStyleSheet()['Title']), t])
    return send_file(path, as_attachment=True)

@app.route('/logout')
def logout(): session.clear(); return redirect('/login')
